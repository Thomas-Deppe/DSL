init 
idle

Timer_Instruct

func Choose_DISP

Mouse_Instruct

lw A A0
sw A 01

lw B A1 
sw B 02

lw A A2
sw A 03

// sw A D0

Choose_DISP

lw A E0
lw B 04
beq A B DISP_X
bltq A B DISP_Status
bgtq A B DISP_Y
goto Mouse_Instruct

DISP_Status
lw A 01
sw A D0
ret

DISP_X
lw A 02
sw A D0
ret

DISP_Y
lw A 03
sw A D0
ret

// Ram allocation:
// 0x00 - Current value to display on seven segment display
// 0x01 - Mouse Status
// 0x02 - MouseX 
// 0x03 - MouseY
// 0x04 - Choose Display
